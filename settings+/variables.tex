% ----------------------------------------------------------------------------
% Figure path utility
% ----------------------------------------------------------------------------
% This macro defines \figpath based on optional and required arguments.
% Usage:
%   \setfigpath{chapter}           → sets \figpath to \home/images/chapter/
%   \setfigpath{chapter}{section}  → sets \figpath to \home/images/chapter/section/
%
% Example:
%   \setfigpath{chapter1}
%   \includegraphics{\figpath diagram.pdf}
% ----------------------------------------------------------------------------
\newcommand{\setfigpath}[2][]{%
  \if\relax\detokenize{#1}\relax
    \def\figpath{\home/images/#2/}%
  \else
    \def\figpath{\home/images/#1/#2/}%
  \fi
}
