                                           & \@firstExaminerDepartment\\
                                           & \@firstExaminerChair\\

                                               & \@secondExaminerDepartment\\
                                               & \@secondExaminerChair\\

                                               & \\
                                               & \\
                                           \fi

                                           & \\

                                           & \@supervisorDepartment\\
                                           & \@supervisorChair\\

                                           &

                \par\noindent\makebox[5cm][l]{\placeDate} \hfill\makebox[5cm][l]{\signature}%
